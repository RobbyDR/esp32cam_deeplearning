# 6) Imbalance report function
def imbalance_report(samples):
    labels = [s[1] for s in samples]
    counts = Counter(labels)
    counts_list = [counts[i] for i in range(len(classes))]
    arr = np.array(counts_list)
    total = arr.sum()
    mean = arr.mean()
    std = arr.std()
    mn = arr.min()
    mx = arr.max()
    imbalance_ratio = mx / mn if mn>0 else np.inf
    cv = std/mean if mean>0 else np.nan
    # normalized entropy
    ps = arr/arr.sum()
    entropy = -np.sum([p*np.log2(p) for p in ps if p>0])
    max_entropy = np.log2(len(arr))
    normalized_entropy = entropy / max_entropy
    df = pd.DataFrame({
        "class_idx": list(range(len(classes))),
        "class_name": classes,
        "count": counts_list
    })
    summary = {
        "total": int(total),
        "mean": float(mean),
        "std": float(std),
        "min": int(mn),
        "max": int(mx),
        "imbalance_ratio_max_min": float(imbalance_ratio),
        "cv": float(cv),
        "entropy": float(entropy),
        "normalized_entropy": float(normalized_entropy)
    }
    return df, summary

# 9) Define Classic CNN (4 conv layers)
class ClassicCNN(nn.Module):
    def __init__(self, num_classes):
        super().__init__()
        self.conv1 = nn.Sequential(nn.Conv2d(3, 16, 3, padding=1), nn.ReLU(), nn.MaxPool2d(2))
        self.conv2 = nn.Sequential(nn.Conv2d(16,32,3,padding=1), nn.ReLU(), nn.MaxPool2d(2))
        self.conv3 = nn.Sequential(nn.Conv2d(32,64,3,padding=1), nn.ReLU(), nn.MaxPool2d(2))
        self.conv4 = nn.Sequential(nn.Conv2d(64,128,3,padding=1), nn.ReLU(), nn.MaxPool2d(2))
        self.flatten_dim = 128 * (CROP_SIZE//16) * (CROP_SIZE//16)  # 64->4
        self.fc = nn.Sequential(nn.Linear(self.flatten_dim, 256), nn.ReLU(), nn.Dropout(0.2), nn.Linear(256, num_classes))
    def forward(self,x):
        x = self.conv1(x); x = self.conv2(x); x = self.conv3(x); x = self.conv4(x)
        x = x.view(x.size(0), -1)
        return self.fc(x)

from sklearn.metrics import f1_score
def evaluate(model, loader):
    model.eval()
    ys, preds = [], []
    losses = []
    with torch.no_grad():
        for X,y in loader:
            X,y = X.to(device), y.to(device)
            out = model(X)
            loss = criterion(out,y)
            losses.append(loss.item() * X.size(0))
            preds.extend(out.argmax(1).cpu().numpy().tolist())
            ys.extend(y.cpu().numpy().tolist())
    avg_loss = sum(losses)/len(loader.dataset)
    acc = (np.array(ys) == np.array(preds)).mean()
    f1_macro = f1_score(ys, preds, average='macro', zero_division=0)
    return avg_loss, acc, f1_macro, ys, preds

        "min": mn, "max": mx, "imbalance_ratio_max_min": ir,

# 8. TinyESP32CNN model (3 depthwise blocks, FC 2048->64->num_classes)
class TinyESP32CNN(nn.Module):
    def __init__(self, num_classes):
        super().__init__()
        def depthwise_pw(in_ch, out_ch):
            return nn.Sequential(
                nn.Conv2d(in_ch, in_ch, kernel_size=3, padding=1, groups=in_ch, bias=False), # depthwise
                nn.Conv2d(in_ch, out_ch, kernel_size=1, bias=False),                          # pointwise
                nn.ReLU(inplace=True),
                nn.MaxPool2d(2,2)
            )
        self.features = nn.Sequential(
            depthwise_pw(3, 8),    # 64 -> 32
            depthwise_pw(8, 16),   # 32 -> 16
            depthwise_pw(16, 32),  # 16 -> 8
        )
        # flatten: 32 * 8 * 8 = 2048
        self.classifier = nn.Sequential(
            nn.Linear(32*8*8, 64),
            nn.ReLU(),
            nn.Dropout(0.1),
            nn.Linear(64, num_classes)
        )
    def forward(self, x):
        x = self.features(x)
        x = x.view(x.size(0), -1)
        x = self.classifier(x)
        return x

def evaluate_loader(model, loader, device, return_preds=False):
    model.eval()
    ys, preds, losses = [], [], []
    with torch.no_grad():
        for X,y in loader:
            X,y = X.to(device), y.to(device)
            out = model(X)
            loss = criterion(out, y)
            losses.append(loss.item() * X.size(0))
            p = out.argmax(1).cpu().numpy()
            preds.extend(p.tolist()); ys.extend(y.cpu().numpy().tolist())
    avg_loss = sum(losses)/len(loader.dataset)
    acc = (np.array(ys) == np.array(preds)).mean()
    f1_macro = f1_score(ys, preds, average='macro', zero_division=0)
    if return_preds:
        return avg_loss, acc, f1_macro, ys, preds
    return avg_loss, acc, f1_macro

def compute_report(ys, preds, classes):

        optimizer.zero_grad()
        loss.backward()
        # grad norm
        grad_norm = 0.0
        for p in model.parameters():
            if p.grad is not None:
                grad_norm += p.grad.data.norm(2).item() ** 2
        grad_norm = math.sqrt(grad_norm)
        grad_norms.append(grad_norm)
        optimizer.step()

        running_loss += loss.item() * X.size(0)
        running_correct += (out.argmax(1) == y).sum().item()
        n_samples += X.size(0)
        batch_times.append(time.time()-t0)

def imbalance_report(samples):
    labels = [s[1] for s in samples]
    counts = Counter(labels)
    counts_list = [counts[i] for i in range(len(classes))]
    arr = np.array(counts_list)
    total = int(arr.sum())
    mean = float(arr.mean()); std = float(arr.std()); mn = int(arr.min()); mx = int(arr.max())
    imbalance_ratio = float(mx / mn) if mn>0 else float('inf')
    cv = float(std/mean) if mean>0 else float('nan')
    ps = arr/arr.sum()
    entropy = -float(np.sum([p*np.log2(p) for p in ps if p>0]))

    return df, summary

# 8) Define A2 TinyESP32CNN (Depthwise ALL)
class TinyA2_DepthwiseAll(nn.Module):
    def __init__(self, num_classes):
        super().__init__()
        # helper: depthwise + pointwise + act + pool
        def depthwise_block(in_ch, out_ch):
            return nn.Sequential(
                nn.Conv2d(in_ch, in_ch, kernel_size=3, padding=1, groups=in_ch, bias=False),  # depthwise
                nn.Conv2d(in_ch, out_ch, kernel_size=1, bias=False),                           # pointwise
                nn.ReLU(inplace=True),
                nn.MaxPool2d(2)
            )
        # channels: 3 -> 6 -> 12 -> 24
        self.features = nn.Sequential(
            depthwise_block(3, 6),    # 64 -> 32
            depthwise_block(6, 12),   # 32 -> 16
            depthwise_block(12, 24)   # 16 -> 8
        )
        # flatten: 24 * 8 * 8 = 1536
        self.classifier = nn.Sequential(
            nn.Linear(24 * 8 * 8, 32),
            nn.ReLU(),
            nn.Dropout(0.1),
            nn.Linear(32, num_classes)
        )
    def forward(self, x):
        x = self.features(x)
        x = x.view(x.size(0), -1)
        x = self.classifier(x)
        return x

def evaluate(model, loader):
    model.eval()
    ys, preds = [], []
    losses = []
    with torch.no_grad():
        for X,y in loader:
            X,y = X.to(device), y.to(device)
            out = model(X)
            loss = criterion(out, y)
            losses.append(loss.item() * X.size(0))
            preds.extend(out.argmax(1).cpu().numpy().tolist())
            ys.extend(y.cpu().numpy().tolist())
    avg_loss = sum(losses) / len(loader.dataset)
    acc = (np.array(ys) == np.array(preds)).mean()
    f1_macro = f1_score(ys, preds, average='macro', zero_division=0)
    return avg_loss, acc, f1_macro, ys, preds

# simplified inverted residual block with optional expansion
class InvertedResidualLite(nn.Module):
    def __init__(self, in_c, out_c, stride=1, expand=4):
        super().__init__()
        mid = max(1, int(in_c * expand))
        # pointwise expand
        if expand > 1:
            self.expand_pw = nn.Sequential(
                nn.Conv2d(in_c, mid, 1, bias=False),
                nn.BatchNorm2d(mid),
                nn.ReLU(inplace=True)
            )
        else:
            self.expand_pw = nn.Identity()
        # depthwise
        self.dw = nn.Sequential(
            nn.Conv2d(mid, mid, 3, stride=stride, padding=1, groups=mid, bias=False),
            nn.BatchNorm2d(mid),
            nn.ReLU(inplace=True)
        )
        # project
        self.project_pw = nn.Sequential(
            nn.Conv2d(mid, out_c, 1, bias=False),
            nn.BatchNorm2d(out_c)
        )
        self.use_res_connect = (stride==1 and in_c==out_c)
    def forward(self, x):
        out = self.expand_pw(x)
        out = self.dw(out)
        out = self.project_pw(out)
        if self.use_res_connect:
            return x + out
        return out

class TinyMobileNetV2Opt(nn.Module):
    def __init__(self, num_classes, alpha=0.3):
        super().__init__()
        def ch(x): return max(1, int(x * alpha))
        # initial conv
        input_ch = ch(32)
        self.stem = nn.Sequential(
            nn.Conv2d(3, input_ch, 3, stride=2, padding=1, bias=False),
            nn.BatchNorm2d(input_ch),
            nn.ReLU(inplace=True)
        )
        # blocks: (in,out,stride,expand)
        blocks_cfg = [
            (input_ch, ch(16), 1, 1),
            (ch(16), ch(24), 2, 4),
            (ch(24), ch(24), 1, 4),
            (ch(24), ch(32), 2, 4),
            (ch(32), ch(32), 1, 4),
            (ch(32), ch(64), 2, 4),
            (ch(64), ch(64), 1, 4)
        ]
        layers=[]
        for in_c,out_c,stride,exp in blocks_cfg:
            layers.append(InvertedResidualLite(in_c,out_c,stride,expand=exp))
        self.features = nn.Sequential(*layers)
        # global pooling
        self.pool = nn.AdaptiveAvgPool2d(1)
        final_ch = ch(64)
        # classifier (kept small)
        self.classifier = nn.Sequential(
            nn.Linear(final_ch, 128),
            nn.ReLU(),
            nn.Dropout(0.2),
            nn.Linear(128, num_classes)
        )
    def forward(self, x):
        x = self.stem(x)
        x = self.features(x)
        x = self.pool(x).reshape(x.size(0), -1)
        x = self.classifier(x)
        return x

def evaluate(loader):
    model.eval()
    ys, preds = [], []
    losses = []
    with torch.no_grad():
        for X,y in loader:
            X,y = X.to(device), y.to(device)
            out = model(X)
            loss = criterion(out,y)
            losses.append(loss.item()*X.size(0))
            preds.extend(out.argmax(1).cpu().numpy().tolist())
            ys.extend(y.cpu().numpy().tolist())
    avg_loss = sum(losses)/len(loader.dataset)
    acc = (np.array(ys)==np.array(preds)).mean()
    f1_macro = f1_score(ys, preds, average='macro', zero_division=0)
    return avg_loss, acc, f1_macro, ys, preds

# simplified inverted residual block with optional expansion
class InvertedResidualLite(nn.Module):
    def __init__(self, in_c, out_c, stride=1, expand=4):
        super().__init__()
        mid = max(1, int(in_c * expand))
        # pointwise expand
        if expand > 1:
            self.expand_pw = nn.Sequential(
                nn.Conv2d(in_c, mid, 1, bias=False),
                nn.BatchNorm2d(mid),
                nn.ReLU(inplace=True)
            )
        else:
            self.expand_pw = nn.Identity()
        # depthwise
        self.dw = nn.Sequential(
            nn.Conv2d(mid, mid, 3, stride=stride, padding=1, groups=mid, bias=False),
            nn.BatchNorm2d(mid),
            nn.ReLU(inplace=True)
        )
        # project
        self.project_pw = nn.Sequential(
            nn.Conv2d(mid, out_c, 1, bias=False),
            nn.BatchNorm2d(out_c)
        )
        self.use_res_connect = (stride==1 and in_c==out_c)
    def forward(self, x):
        out = self.expand_pw(x)
        out = self.dw(out)
        out = self.project_pw(out)
        if self.use_res_connect:
            return x + out
        return out

class TinyMobileNetV2Opt(nn.Module):
    def __init__(self, num_classes, alpha=0.3):
        super().__init__()
        def ch(x): return max(1, int(x * alpha))
        # initial conv
        input_ch = ch(32)
        self.stem = nn.Sequential(
            nn.Conv2d(3, input_ch, 3, stride=2, padding=1, bias=False),
            nn.BatchNorm2d(input_ch),
            nn.ReLU(inplace=True)
        )
        # blocks: (in,out,stride,expand)
        blocks_cfg = [
            (input_ch, ch(16), 1, 1),
            (ch(16), ch(24), 2, 4),
            (ch(24), ch(24), 1, 4),
            (ch(24), ch(32), 2, 4),
            (ch(32), ch(32), 1, 4),
            (ch(32), ch(64), 2, 4),
            (ch(64), ch(64), 1, 4)
        ]
        layers=[]
        for in_c,out_c,stride,exp in blocks_cfg:
            layers.append(InvertedResidualLite(in_c,out_c,stride,expand=exp))
        self.features = nn.Sequential(*layers)
        # global pooling
        self.pool = nn.AdaptiveAvgPool2d(1)
        final_ch = ch(64)
        # classifier (kept small)
        self.classifier = nn.Sequential(
            nn.Linear(final_ch, 128),
            nn.ReLU(),
            nn.Dropout(0.2),
            nn.Linear(128, num_classes)
        )
    def forward(self, x):
        x = self.stem(x)
        x = self.features(x)
        x = self.pool(x).reshape(x.size(0), -1)
        x = self.classifier(x)
        return x

    return nn.Sequential(
        nn.Conv2d(inp, oup, 3, stride, 1, bias=False),
        nn.BatchNorm2d(oup),
        nn.ReLU(inplace=True)

    return nn.Sequential(
        nn.Conv2d(inp, inp, 3, stride, 1, groups=inp, bias=False),
        nn.BatchNorm2d(inp),
        nn.ReLU(inplace=True),

            dw_pw(48, 64, 1),

            nn.Linear(flat, 128),
            nn.ReLU(True),
            nn.Dropout(0.2),
            nn.Linear(128, num_classes)

# 6. Imbalance report (for printing)
def imbalance_report(samples):
    labels = [s[1] for s in samples]
    counts = Counter(labels)
    counts_list = [counts.get(i,0) for i in range(num_classes)]
    arr = np.array(counts_list)
    total = int(arr.sum())
    mean = float(arr.mean())
    std = float(arr.std())
    mn = int(arr.min())
    mx = int(arr.max())
    imbalance_ratio = float(mx / mn) if mn>0 else float('inf')
    cv = float(std / mean) if mean>0 else float('nan')
    ps = arr / arr.sum()
    entropy = -float(np.sum([p*np.log2(p) for p in ps if p>0]))
    max_entropy = float(np.log2(len(arr)))
    normalized_entropy = entropy / max_entropy if max_entropy>0 else 0.0
    df = pd.DataFrame({"class_idx":list(range(num_classes)),"class_name":classes,"count":counts_list})
    summary = {"total":total,"mean":mean,"std":std,"min":mn,"max":mx,"imbalance_ratio_max_min":imbalance_ratio,"cv":cv,"entropy":entropy,"normalized_entropy":normalized_entropy}
    return df, summary

# 8. Define ESPNet (Ultra-small) - custom tiny design
# Design goals:
# - depthwise separable ops where possible
# - small channel counts
# - output flatten small (24*8*8 -> etc)
# This is tiny; number of params kept very low.

def conv_bn_relu(in_c, out_c, k=3, s=1, p=1):
    return nn.Sequential(nn.Conv2d(in_c, out_c, kernel_size=k, stride=s, padding=p, bias=False),
                         nn.BatchNorm2d(out_c),
                         nn.ReLU(inplace=True))

def dw_pw(in_c, out_c, stride=1):
    # depthwise then pointwise, minimal BN
    return nn.Sequential(
        nn.Conv2d(in_c, in_c, 3, stride=stride, padding=1, groups=in_c, bias=False),
        nn.BatchNorm2d(in_c),
        nn.ReLU(inplace=True),
        nn.Conv2d(in_c, out_c, 1, bias=False),
        nn.BatchNorm2d(out_c),
        nn.ReLU(inplace=True)
    )

class ESPBlockTiny(nn.Module):
    """A tiny ESP-like block: split->dilated convs approximated using small depthwise branches
       For extreme smallness we keep only a couple of branches and then fuse."""
    def __init__(self, in_c, out_c):
        super().__init__()
        mid = max(1, out_c//2)
        # reduce
        self.reduce = nn.Conv2d(in_c, mid, 1, bias=False)
        # two small depthwise branches (simulate wider receptive fields cheaply)
        self.dw1 = nn.Conv2d(mid, mid, 3, padding=1, groups=mid, bias=False)
        self.dw2 = nn.Conv2d(mid, mid, 3, padding=1, groups=mid, bias=False)
        self.bn = nn.BatchNorm2d(mid*2)
        self.project = nn.Conv2d(mid*2, out_c, 1, bias=False)
        self.act = nn.ReLU(inplace=True)
    def forward(self, x):
        r = self.reduce(x)
        a = self.dw1(r)
        b = self.dw2(r)
        cat = torch.cat([a,b], dim=1)
        cat = self.bn(cat)
        out = self.project(cat)
        return self.act(out)

class ESPNetUltra(nn.Module):
    def __init__(self, num_classes):
        super().__init__()
        # extremely small channel plan
        # input 3x64x64
        self.stem = conv_bn_relu(3, 8, k=3, s=2, p=1)   # 64 -> 32
        self.esp1 = ESPBlockTiny(8, 12)                  # 32 -> 32
        self.down1 = dw_pw(12, 16, stride=2)             # 32 -> 16
        self.esp2 = ESPBlockTiny(16, 20)                 # 16 -> 16
        self.down2 = dw_pw(20, 24, stride=2)             # 16 -> 8
        self.esp3 = ESPBlockTiny(24, 24)                 # 8 -> 8

        # classifier: make flatten very small
        # flatten dim = 24 * 8 * 8 = 1536
        self.pool = nn.AdaptiveAvgPool2d((1,1))  # reduce spatial
        self.fc_small = nn.Sequential(
            nn.Flatten(),
            nn.Linear(24, 32),   # tiny bottleneck
            nn.ReLU(inplace=True),
            nn.Dropout(0.1),
            nn.Linear(32, num_classes)
        )

    def forward(self, x):
        x = self.stem(x)
        x = self.esp1(x)
        x = self.down1(x)
        x = self.esp2(x)
        x = self.down2(x)
        x = self.esp3(x)
        x = self.pool(x)   # [B,24,1,1]
        return self.fc_small(x)

# 11. Evaluation helper
def evaluate_model(m, loader):
    m.eval()
    ys = []; preds = []; losses = []
    with torch.no_grad():
        for X,y in loader:
            X,y = X.to(DEVICE), y.to(DEVICE)
            out = m(X)
            loss = criterion(out, y)
            losses.append(loss.item() * X.size(0))
            preds.extend(out.argmax(1).cpu().numpy().tolist())
            ys.extend(y.cpu().numpy().tolist())
    avg_loss = sum(losses) / len(loader.dataset)
    acc = (np.array(ys) == np.array(preds)).mean()
    f1m = f1_score(ys, preds, average='macro', zero_division=0)
    return avg_loss, acc, f1m, ys, preds

# 12. Training loop (with logging, early stopping on val macro-F1)
best_val_f1 = 0.0
patience = 0
history = {"train_loss":[], "train_acc":[], "val_loss":[], "val_acc":[], "val_f1":[]}

start_time = time.time()
for epoch in range(1, MAX_EPOCHS+1):
    model.train()
    running_loss = 0.0; running_correct = 0; total = 0
    batch_times = []
    for X,y in tqdm(train_loader, desc=f"Epoch {epoch} train"):
        t0 = time.time()
        X,y = X.to(DEVICE), y.to(DEVICE)
        out = model(X)
        loss = criterion(out, y)
        optimizer.zero_grad(); loss.backward(); optimizer.step()
        batch_times.append(time.time()-t0)

        running_loss += loss.item() * X.size(0)
        running_correct += (out.argmax(1) == y).sum().item()
        total += X.size(0)

    train_loss = running_loss / total
    train_acc = running_correct / total
    val_loss, val_acc, val_f1, _, _ = evaluate_model(model, valid_loader)

    history["train_loss"].append(train_loss)
    history["train_acc"].append(train_acc)
    history["val_loss"].append(val_loss)
    history["val_acc"].append(val_acc)
    history["val_f1"].append(val_f1)

    print(f"Epoch {epoch:02d} | train_loss {train_loss:.4f} acc {train_acc:.4f} | val_loss {val_loss:.4f} acc {val_acc:.4f} f1 {val_f1:.4f} | avg_batch_s {np.mean(batch_times):.3f}")

    scheduler.step(val_f1)

    # save best
    if val_f1 > best_val_f1 + 1e-4:
        best_val_f1 = val_f1
        torch.save(model.state_dict(), os.path.join(MODEL_SAVE_DIR, "ESPNetUltra_best.pth"))
        print("  -> New best saved (val_f1 improved).")
        patience = 0
    else:
        patience += 1

    if patience >= EARLY_STOPPING:
        print("Early stopping triggered.")
        break

end_time = time.time()
print("Training finished in", round(end_time-start_time,1), "s")
print("Best val F1:", best_val_f1)

def show_misclassified(n=18):
    bad_idx = [i for i,(t,p) in enumerate(zip(ys_true, ys_pred)) if t!=p]
    if not bad_idx:
        print("No misclassified.")
        return
    picks = bad_idx[:n]
    imgs=[]; titles=[]
    for idx in picks:
        path, true_label = test_set.samples[idx]
        pil = test_set.loader(path)
        img = eval_transform(pil)
        imgs.append(img)
        titles.append(f"T:{classes[true_label]} P:{classes[ys_pred[idx]]}")
    grid = utils.make_grid(imgs, nrow=6, normalize=True, scale_each=True)
    plt.figure(figsize=(12,8)); plt.imshow(np.transpose(grid.numpy(), (1,2,0))); plt.axis('off'); plt.title("Misclassified samples")
    for t in titles[:n]:
        print(t)

# 17. Feature map preview (stem)
def visualize_activations(sample_idx=0, n_maps=8):
    model.eval()
    path, label = test_set.samples[sample_idx]
    pil = test_set.loader(path)
    inp = eval_transform(pil).unsqueeze(0).to(DEVICE)
    with torch.no_grad():
        _ = model(inp)
    if 'stem' in activations:
        acts = activations['stem'].squeeze(0)  # [C,H,W]
        n_maps = min(n_maps, acts.shape[0])
        fig, axes = plt.subplots(1, n_maps, figsize=(n_maps*2,2))
        for i in range(n_maps):
            axes[i].imshow(acts[i].numpy(), cmap='viridis'); axes[i].axis('off')
        plt.suptitle(f"Stem feature maps for sample {sample_idx} (label={classes[label]})")
        plt.show()
    else:
        print("No activation captured.")
