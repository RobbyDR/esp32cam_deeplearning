// step B3
// framesize 96x96

#define PIN_SD_CLK 39
#define PIN_SD_CMD 38
#define PIN_SD_D0 40

#include <Arduino.h>
#include "camera.h"
#include "storage.h"
#include "dataset.h"

bool auto_mode = false;
String auto_class = "";
uint32_t auto_interval = 1000;
uint32_t auto_total = 0;
uint32_t auto_counter = 0;
uint32_t auto_last_time = 0;

void execute_capture(const String &class_name)
{
    if (!storage_ready())
    {
        Serial.println("SD not ready");
        return;
    }

    if (!camera_capture_raw())
    {
        Serial.println("Capture failed");
        return;
    }

    uint32_t id = camera_frame_id();

    char folder[64];
    snprintf(folder, sizeof(folder), "/sdcard/%s", class_name.c_str());
    mkdir(folder, 0775);

    char raw_path[128];
    char jpg_path[128];
    char json_path[128];

    snprintf(raw_path, sizeof(raw_path), "%s/%lu.raw", folder, id);
    snprintf(jpg_path, sizeof(jpg_path), "%s/%lu.jpg", folder, id);
    snprintf(json_path, sizeof(json_path), "%s/%lu.json", folder, id);

    /* RAW */
    FILE *fraw = fopen(raw_path, "wb");
    if (fraw)
    {
        fwrite(camera_raw_buffer(), 1, camera_raw_length(), fraw);
        fclose(fraw);
    }

    /* JPEG preview */
    camera_fb_t *fb = camera_raw_to_jpeg();
    if (fb)
    {
        FILE *fjpg = fopen(jpg_path, "wb");
        if (fjpg)
        {
            fwrite(fb->buf, 1, fb->len, fjpg);
            fclose(fjpg);
        }
        camera_release(fb);
    }

    /* METADATA */
    FILE *fjson = fopen(json_path, "w");
    if (fjson)
    {
        fprintf(fjson,
                "{\n"
                "  \"frame_id\": %lu,\n"
                "  \"class\": \"%s\",\n"
                "  \"width\": %u,\n"
                "  \"height\": %u,\n"
                "  \"avg_diff\": %lu,\n"
                "  \"confidence\": %u,\n"
                "  \"millis\": %lu\n"
                "}\n",
                id,
                class_name.c_str(),
                camera_width(),
                camera_height(),
                camera_avg_diff(),
                camera_confidence(),
                millis());
        fclose(fjson);
    }

    Serial.printf("Saved frame %lu (%s)\n", id, class_name.c_str());
}

void handle_serial()
{
    if (!Serial.available())
        return;

    String cmd = Serial.readStringUntil('\n');
    cmd.trim();

    if (cmd.startsWith("capture_auto"))
    {
        // format: capture_auto wereng 1000 50
        int p1 = cmd.indexOf(' ');
        int p2 = cmd.indexOf(' ', p1 + 1);
        int p3 = cmd.indexOf(' ', p2 + 1);

        if (p1 < 0 || p2 < 0 || p3 < 0)
        {
            Serial.println("Format: capture_auto <class> <interval_ms> <count>");
            return;
        }

        auto_class = cmd.substring(p1 + 1, p2);
        auto_interval = cmd.substring(p2 + 1, p3).toInt();
        auto_total = cmd.substring(p3 + 1).toInt();

        auto_counter = 0;
        auto_last_time = millis();
        auto_mode = true;

        Serial.printf("AUTO START: %s, interval=%lu ms, total=%lu\n",
                      auto_class.c_str(), auto_interval, auto_total);
    }
    else if (cmd.startsWith("capture"))
    {
        int p = cmd.indexOf(' ');
        if (p < 0)
        {
            Serial.println("Format: capture <class>");
            return;
        }

        String class_name = cmd.substring(p + 1);
        execute_capture(class_name);
    }
    else if (cmd == "stop")
    {
        auto_mode = false;
        Serial.println("AUTO STOPPED");
    }
}

void setup()
{
    Serial.begin(115200);
    delay(1000);

    Serial.println("Dataset Builder Mode (No WiFi)");

    if (!camera_init())
    {
        Serial.println("Camera init failed");
        while (true)
            ;
    }

    if (!storage_init(PIN_SD_CLK, PIN_SD_CMD, PIN_SD_D0))
    {
        Serial.println("SD init failed");
        while (true)
            ;
    }

    Serial.println("Ready. Type: capture <class>");
}

void loop()
{
    handle_serial();

    if (auto_mode)
    {
        if (millis() - auto_last_time >= auto_interval)
        {
            auto_last_time = millis();

            execute_capture(auto_class);
            auto_counter++;

            if (auto_counter >= auto_total)
            {
                auto_mode = false;
                Serial.println("AUTO COMPLETE");
            }
        }
    }
}

#include "camera.h"
#include <Arduino.h>
#include "img_converters.h"

static const int W = 96;
static const int H = 96;
static const int PIXELS = W * H;
static const uint32_t THRESHOLD = 20;

static uint8_t *curr = NULL;
static uint8_t *prev = NULL;

static uint32_t avgDiff = 0;
static uint8_t confidence = 0;
static bool detected = false;

static uint32_t frame_id = 0;
static uint32_t frame_size = 0;

uint16_t camera_width() { return W; }
uint16_t camera_height() { return H; }

/* ========= INIT ========= */
bool camera_init()
{
    camera_config_t c = {};
    c.ledc_channel = LEDC_CHANNEL_0;
    c.ledc_timer = LEDC_TIMER_0;

    c.pin_d0 = 11;
    c.pin_d1 = 9;
    c.pin_d2 = 8;
    c.pin_d3 = 10;
    c.pin_d4 = 12;
    c.pin_d5 = 18;
    c.pin_d6 = 17;
    c.pin_d7 = 16;

    c.pin_xclk = 15;
    c.pin_pclk = 13;
    c.pin_vsync = 6;
    c.pin_href = 7;
    c.pin_sccb_sda = 4;
    c.pin_sccb_scl = 5;

    c.pin_pwdn = -1;
    c.pin_reset = -1;

    c.xclk_freq_hz = 20000000;
    c.pixel_format = PIXFORMAT_GRAYSCALE;
    c.frame_size = FRAMESIZE_96X96;
    c.fb_count = 2;
    c.grab_mode = CAMERA_GRAB_LATEST;

    if (esp_camera_init(&c) != ESP_OK)
        return false;

    curr = (uint8_t *)malloc(PIXELS);
    prev = (uint8_t *)malloc(PIXELS);
    memset(prev, 0, PIXELS);

    return true;
}

/* ========= CAPTURE RAW ========= */
bool camera_capture_raw()
{
    // flush 1 frame lama
    camera_fb_t *fb_flush = esp_camera_fb_get();
    if (fb_flush)
        esp_camera_fb_return(fb_flush);

    delay(30); // beri waktu sensor update

    camera_fb_t *fb = esp_camera_fb_get();
    if (!fb)
        return false;

    frame_id++;
    frame_size = fb->len;

    memcpy(curr, fb->buf, PIXELS);

    // hitung difference hanya jika bukan frame pertama
    if (frame_id > 1)
    {
        uint32_t sum = 0;

        for (int i = 0; i < PIXELS; i++)
            sum += abs((int)curr[i] - (int)prev[i]);

        avgDiff = sum / PIXELS;

        uint32_t scaled = (avgDiff * 100) / THRESHOLD;

        if (avgDiff >= THRESHOLD)
        {
            detected = true;
            confidence = 100;
        }
        else
        {
            detected = false;
            confidence = scaled > 100 ? 100 : scaled;
        }
    }
    else
    {
        avgDiff = 0;
        confidence = 0;
        detected = false;
    }

    memcpy(prev, curr, PIXELS);

    esp_camera_fb_return(fb);

    return true;
}

/* ========= RAW → JPEG ========= */
camera_fb_t *camera_raw_to_jpeg()
{
    if (!curr || frame_id == 0)
        return NULL;

    camera_fb_t *fb = (camera_fb_t *)malloc(sizeof(camera_fb_t));
    uint8_t *jpeg_buf = NULL;
    size_t jpeg_len = 0;

    bool ok = fmt2jpg(curr, PIXELS, W, H,
                      PIXFORMAT_GRAYSCALE,
                      80, &jpeg_buf, &jpeg_len);

    if (!ok)
        return NULL;

    fb->buf = jpeg_buf;
    fb->len = jpeg_len;
    fb->format = PIXFORMAT_JPEG;
    return fb;
}

/* ========= GETTERS ========= */
bool camera_object_detected() { return detected; }
uint32_t camera_avg_diff() { return avgDiff; }
uint8_t camera_confidence() { return confidence; }
uint32_t camera_frame_size() { return frame_size; }
uint32_t camera_frame_id() { return frame_id; }
uint32_t camera_threshold() { return THRESHOLD; }

void camera_release(camera_fb_t *fb)
{
    free(fb->buf);
    free(fb);
}

uint8_t *camera_raw_buffer() { return curr; }
uint32_t camera_raw_length() { return PIXELS; }

#include "dataset.h"
#include "camera.h"
#include "storage.h"
#include <sys/stat.h>
#include <time.h>

static uint32_t global_id = 0;

static void ensure_directory(const char *path)
{
    struct stat st;
    if (stat(path, &st) != 0)
    {
        mkdir(path, 0755);
    }
}

bool dataset_capture(const char *class_name)
{
    if (!storage_ready())
    {
        Serial.println("{\"error\":\"sd_not_ready\"}");
        return false;
    }

    if (!camera_capture_raw())
    {
        Serial.println("{\"error\":\"capture_failed\"}");
        return false;
    }

    global_id++;

    char class_path[128];
    snprintf(class_path, sizeof(class_path), "/sdcard/%s", class_name);
    ensure_directory(class_path);

    char raw_path[160];
    char jpg_path[160];
    char json_path[160];

    snprintf(raw_path, sizeof(raw_path), "%s/%06lu.raw", class_path, global_id);
    snprintf(jpg_path, sizeof(jpg_path), "%s/%06lu.jpg", class_path, global_id);
    snprintf(json_path, sizeof(json_path), "%s/%06lu.json", class_path, global_id);

    bool raw_ok = false;
    bool jpg_ok = false;
    bool json_ok = false;

    /* ===== SAVE RAW ===== */
    FILE *fraw = fopen(raw_path, "wb");
    if (fraw)
    {
        fwrite(camera_raw_buffer(), 1, camera_raw_length(), fraw);
        fclose(fraw);
        raw_ok = true;
    }

    /* ===== SAVE JPEG ===== */
    camera_fb_t *fb = camera_raw_to_jpeg();
    if (fb)
    {
        FILE *fjpg = fopen(jpg_path, "wb");
        if (fjpg)
        {
            fwrite(fb->buf, 1, fb->len, fjpg);
            fclose(fjpg);
            jpg_ok = true;
        }
        camera_release(fb);
    }

    /* ===== SAVE METADATA ===== */
    FILE *fjson = fopen(json_path, "w");
    if (fjson)
    {
        time_t now = time(NULL);

        fprintf(fjson,
                "{\n"
                "  \"id\": %lu,\n"
                "  \"class\": \"%s\",\n"
                "  \"width\": %d,\n"
                "  \"height\": %d,\n"
                "  \"raw_length\": %d,\n"
                "  \"timestamp\": %ld\n"
                "}\n",
                global_id,
                class_name,
                camera_width(),
                camera_height(),
                camera_raw_length(),
                now);

        fclose(fjson);
        json_ok = true;
    }

    /* ===== PRINT STATUS ===== */
    Serial.printf(
        "{ \"id\": %lu, \"class\": \"%s\", \"raw\": %s, \"jpg\": %s, \"json\": %s }\n",
        global_id,
        class_name,
        raw_ok ? "true" : "false",
        jpg_ok ? "true" : "false",
        json_ok ? "true" : "false");

    return (raw_ok && jpg_ok && json_ok);
}

storage.cpp
#include "storage.h"
#include "esp_vfs_fat.h"
#include "sdmmc_cmd.h"
#include "driver/sdmmc_host.h"

static sdmmc_card_t *sdcard;
static bool sd_ready = false;

bool storage_init(int pin_clk, int pin_cmd, int pin_d0)
{
    sdmmc_host_t host = SDMMC_HOST_DEFAULT();
    host.flags = SDMMC_HOST_FLAG_1BIT;
    host.max_freq_khz = SDMMC_FREQ_DEFAULT;

    sdmmc_slot_config_t slot_config = SDMMC_SLOT_CONFIG_DEFAULT();
    slot_config.width = 1;
    slot_config.clk = (gpio_num_t)pin_clk;
    slot_config.cmd = (gpio_num_t)pin_cmd;
    slot_config.d0 = (gpio_num_t)pin_d0;

    esp_vfs_fat_sdmmc_mount_config_t mount_config = {
        .format_if_mount_failed = false,
        .max_files = 5,
        .allocation_unit_size = 16 * 1024};

    esp_err_t ret = esp_vfs_fat_sdmmc_mount("/sdcard",
                                            &host,
                                            &slot_config,
                                            &mount_config,
                                            &sdcard);

    if (ret != ESP_OK)
    {
        Serial.printf("SD Mount Failed: %s\n", esp_err_to_name(ret));
        return false;
    }

    sd_ready = true;
    Serial.println("SD Mounted OK");
    return true;
}

bool storage_ready()
{
    return sd_ready;
}
